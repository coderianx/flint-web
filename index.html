Quick Start

package main

import "github.com/coderianx/flint"

func main() {
    // Create a new Flint application
    app := flint.NewServer()
    
    // Define a simple route
    app.Handle("/", func(ctx *flint.Context) {
        ctx.String(200, "Hello, Flint!")
    })

    // Start the server
    app.Run(":8080")
}

Install with: go get github.com/coderianx/flint
